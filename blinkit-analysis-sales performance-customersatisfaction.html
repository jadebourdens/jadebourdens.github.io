SELECT outlet_size,
       CAST(SUM(Total_sales) AS DECIMAL(10,2)) AS Total_sales,
       CAST((SUM(Total_Sales) * 100.0 / SUM(SUM(Total_Sales)) OVER()) AS DECIMAL(10,2)) AS Sales_Percentage
